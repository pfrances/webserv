server {
	listen 8080;
	server_name webserv;

	root webservPages;
	error_page 404 webservPages/errorPages/404.html;
	client_body_size 1m;

	location / {
		autoindex on;
	}

	location /upload {
		index upload.html;
	}

	location /post {
		allow_methods POST;
		upload_path webservPages/upload;
	}

	location /delete {
		allow_methods DELETE;
		upload_path webservPages/upload;
	}

	location /cgi-bin {
		allow_methods GET POST;
		index cgi.sh;

		cgi_executor /bin/bash;
		cgi_ext .sh;
	}
}
